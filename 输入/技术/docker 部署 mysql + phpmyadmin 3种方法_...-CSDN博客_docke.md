### 方法2: network

	# 创建网络 net-mysql
	# docker network create net-mysql
	
	# 创建容器 test-mysql 并加入网络 net-mysql
	# docker run --restart always --name test-mysql -e MYSQL_USER=root -e MYSQL_PASSWORD=root -e MYSQL_ROOT_PASSWORD=root -p 3306:3306 --network net-mysql -d mysql:5.6

### 方法3: docker-compose.yml

docker-compose.yml 内容:

	version: "2"
	services:
	  mysql:
	    image: mysql:5.6
	    container_name: test-mysql
	    restart: always
	    ports:
	
	      - "3306:3306"
	
	    environment:
	      MYSQL_USER: "root"
	      MYSQL_PASSWORD: "root"
	      MYSQL_ROOT_PASSWORD: "root"
	    networks:
	
	      - net-mysql
	
	  phpmyadmin:
	    image: phpmyadmin/phpmyadmin:latest
	    container_name: test-phpmyadmin
	    ports:
	
	      - "8080:80"
	
	    environment:
	      MYSQL_USER: "root"
	      MYSQL_PASSWORD: "root"
	      MYSQL_ROOT_PASSWORD: "root"
	      PMA_HOST: test-mysql
	    networks:
	
	      - net-mysql
	
	networks:
	  net-mysql:
